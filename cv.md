function highAndLow(numbers){
  let num = numbers.split(" ");
  let min = +num[0];
  let max = +num[0];
  for (let i = 1; i < num.length; i++) {
    if (+num[i] > max) {
      max = +num[i];
    }
    if (+num[i] < min) {
      min = +num[i];
    }
  }
  return max + " " + min;
}
